CREATE VIEW ibk.INFORMACION_VENTAS_VIEW AS
SELECT 
MES,
DOCUMENTOIDENT,
CODIGOSF,
CODIGOCLIENTE,
case when vta_activo2 + VTA_ACTIVO3+ VTA_ACTIVO4 + VTA_ACTIVO5+VTA_ACTIVO7
+VTA_ACTIVO8 +VTA_ACTIVO9 > 0 then 1 else 0 end FLAG_VTA_ACTIVO,
case when VTA_PASIVO13 + VTA_PASIVO14+ VTA_PASIVO3 + VTA_PASIVO4+VTA_PASIVO8 + 
VTA_PASIVO9 > 0 then 1 else 0 end FLAG_VTA_PASIVO,
case when vta_activo2 + VTA_ACTIVO3+ VTA_ACTIVO4 + VTA_ACTIVO5+VTA_ACTIVO7
+VTA_ACTIVO8 +VTA_ACTIVO9 + VTA_PASIVO13 + VTA_PASIVO14+ VTA_PASIVO3 + VTA_PASIVO4+VTA_PASIVO8 + 
VTA_PASIVO9 > 0 then 1 else 0 end FLAG_VTA
FROM ibk.informacion_ventas;
